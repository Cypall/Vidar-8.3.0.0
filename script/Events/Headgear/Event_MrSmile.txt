			mes "with Very Simple Items.";

			mes "Now these are Items needed for `Mr.Smile' the Mask.";

			mes "[Smile Gal]";
			mes "^3355FF10 Jellopy^000000";
			mes "^3355FF10 Fluff^000000";
			mes "^3355FF10 Clover^000000";
			next;
			mes "[Smile Gal]";

			mes "to help Each other and always Smile.";
			close;

		checkitem 909,10,-,noitems;
		checkitem 914,10,-,noitems;
		checkitem 705,10,-,noitems;

			mes "Congratulations !";
			mes "Please take this Mask,`Mr.Smile'.";
			next;
			delitem 909,10;
			delitem 914,10;
			delitem 705,10;

			mes "the Ministry of Information,";
