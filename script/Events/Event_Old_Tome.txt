yuno_in01.vhm,171,94,6	script	Arkyhm	705,{
	check OTQ = 1,SEARCH2,-;
	check OTQ = 2,CONTINUE,-;
	check OTQ = 3,QUEST,-;
	check OTQ = 4,QUEST2,-;
	check OTQ = 5,-,START;
		mes "[Arkyhm]";
		mes "Thank you for all your help!";
	close;
START:
	check OTQ = 6,EXIT2,-;
		mes "[Arkyhm]";
		mes "Hello there.";
		mes "I'm searching for an old tome of spells.";
	next;
		mes "Do you think you can help me?";
	next;
		menu "Of course.",SEARCH,"Sorry.",-;
			mes "Oh... All right.";
	close;
SEARCH:
		mes "[Arkyhm]";
		mes "Thank you so much.";
	goto SEARCH3;
SEARCH2:
	mes "[Arkyhm]";
SEARCH3:
		mes "I've been looking through this library for hours.";
		mes "Perhaps you should try looking around other libraries first.";
		set OTQ = 1;
	close;
CONTINUE:
		mes "[Arkyhm]";
		mes "Oh my... Hm...";
		mes "I think that ^FF00FFReisan^000000 may know of them.";
		mes "He's usually at ^00FF00Alberta^000000.";
	close;
QUEST:
		mes "[Arkyhm]";
		mes "This is terrible.";
		mes "I'm sorry to put you through all this trouble.";
		mes "I don't know this man named ^FF00FFSanyo^000000.";
	close;
QUEST2:
	checkitem 7117,1,-,EXIT;
		mes "Thank you for your help.";
		mes "I know!  I'll give you some status points.";
		set StatusPoint + 20;
		set OTQ = 5;
	next;
		mes "There you go.";
	close;
EXIT:
		mes "[Arkyhm]";
		mes "You did what?";
		set OTQ = 6;
EXIT2:
		mes "I thought you were a good person >=(";
	close;
}
